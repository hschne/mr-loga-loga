def add(severity, message = nil, **context, &block)
  severity ||= UNKNOWN
  return true unless @logger.log?(severity)

  context = merge_context(@context, context)
  context = context.call if context.is_a?(Proc)

  @logger.add(severity, message, **context, &block)
end
